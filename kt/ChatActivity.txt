    override fun onWindowFocusChanged(hasFocus: Boolean) {
        super.onWindowFocusChanged(hasFocus)
        isWindowFocused = hasFocus

    override fun onDestroy() {
        super.onDestroy()
        db.collection("users").document(currentUserId)

    private fun dismissNotifications() {
        val notificationManager = getSystemService(NOTIFICATION_SERVICE)
                as NotificationManager
        notificationManager.cancelAll()
    }

    private fun uploadImageAndSend(imageUri: Uri) {
        Toast.makeText(this, "Uploading image...", Toast.LENGTH_SHORT).show()
        CloudinaryHelper.uploadImage(
            context = this,
            imageUri = imageUri,
            onSuccess = { imageUrl -> sendImageMessage(imageUrl) },
            onError = { error ->
                Toast.makeText(this, "Upload failed: $error", Toast.LENGTH_SHORT).show()
            },
            onProgress = {
                Toast.makeText(this, "Uploading...", Toast.LENGTH_SHORT).show()
            }
        )
    }

    private fun sendImageMessage(imageUrl: String) {
        val messageRef = db.collection("chats")
            .document(chatId)
            .collection("messages")
            .document()

        val message = hashMapOf(
            "messageId" to messageRef.id,
            "senderId" to currentUserId,
            "text" to "",
            "imageUrl" to imageUrl,
            "type" to "image",
            "timestamp" to System.currentTimeMillis(),
            "isRead" to false,
            "status" to "sent"
        )

        messageRef.set(message)
            .addOnSuccessListener {
                messageRef.update("status", "delivered")
                updateLastMessage("📷 Photo", "image")
                sendNotificationToUser(otherUserId, "📷 Photo")
            }
            .addOnFailureListener { e ->
                Toast.makeText(this, "Failed: ${e.message}", Toast.LENGTH_SHORT).show()
            }
    }

    private fun sendTextMessage(text: String) {
        val messageRef = db.collection("chats")
            .document(chatId)
            .collection("messages")
            .document()

        val isOnline = NetworkUtils.isOnline(this)

        // REPLACE with:
        val payload = if (recipientPublicKey.isNotEmpty() && myPublicKey.isNotEmpty()) {
            EncryptionHelper.encryptMessageForBoth(text, recipientPublicKey, myPublicKey)
        } else null
        android.util.Log.d("SLOT", "recipientPublicKey last 20: ${recipientPublicKey.takeLast(20)}")
        android.util.Log.d("SLOT", "myPublicKey last 20: ${myPublicKey.takeLast(20)}")

        android.util.Log.d("KEY_MATCH", "recipientPublicKey last 20: ${recipientPublicKey.takeLast(20)}")
        android.util.Log.d("KEY_MATCH", "encKeyForRecipient last 20: ${payload?.encKeyForRecipient?.takeLast(20)}")

        val message = hashMapOf(
            "messageId" to messageRef.id,
            "senderId" to currentUserId,
            "text" to "",
            "imageUrl" to "",
            "type" to "text",
            "timestamp" to System.currentTimeMillis(),
            "isRead" to false,
            "status" to if (isOnline) "sent" else "pending",
            "encKeyForSender" to (payload?.encKeyForSender ?: ""),
            "encKeyForRecipient" to (payload?.encKeyForRecipient ?: ""),
            "encIv" to (payload?.iv ?: ""),
            "encCiphertext" to (payload?.ciphertext ?: "")
        )

        val localMessage = Message(
            messageId = messageRef.id,
            senderId = currentUserId,
            text = text,
            type = "text",
            timestamp = System.currentTimeMillis(),
            isRead = false,
            status = if (isOnline) "sent" else "pending"
        )
        messageList.add(localMessage)
        adapter.notifyItemInserted(messageList.size - 1)
        recyclerMessages.scrollToPosition(messageList.size - 1)

        messageRef.set(message)
            .addOnSuccessListener {
                messageRef.update("status", "delivered")
                updateLastMessage(text, "text")
                sendNotificationToUser(otherUserId, text)
            }
            .addOnFailureListener {
                Toast.makeText(this, "Failed to send", Toast.LENGTH_SHORT).show()
            }
    }

    private fun markMessagesAsRead() {
        // Reset unread count for current user
        db.collection("chats").document(chatId)
            .update("unreadCount_$currentUserId", 0)

        db.collection("chats")
            .document(chatId)
            .collection("messages")
            .whereEqualTo("senderId", otherUserId)
            .whereEqualTo("isRead", false)
            .get()
            .addOnSuccessListener { documents ->
                for (doc in documents) {
                    doc.reference.update(mapOf("isRead" to true, "status" to "read"))
                }
            }
    }

    private fun updateLastMessage(text: String, type: String) {
        val chatRef = db.collection("chats").document(chatId)
        val lastMessageText = if (type == "image") "📷 Photo" else text

        chatRef.get().addOnSuccessListener { doc ->
            if (doc.exists()) {
                chatRef.update(
                    "lastMessage", lastMessageText,
                    "lastMessageTime", System.currentTimeMillis(),
                    "lastMessageSenderId", currentUserId,
                    "unreadCount_$otherUserId",
                    com.google.firebase.firestore.FieldValue.increment(1)
                )
            } else {
                val data = HashMap<String, Any>()
                data["lastMessage"] = lastMessageText
                data["lastMessageTime"] = System.currentTimeMillis()
                data["lastMessageSenderId"] = currentUserId
                data["user1"] = currentUserId
                data["user2"] = otherUserId
                data["members"] = listOf(currentUserId, otherUserId)
                data["unreadCount_$otherUserId"] = 1
                data["unreadCount_$currentUserId"] = 0
                chatRef.set(data)
            }
        }
    }

    private fun listenForMessages() {
        db.collection("chats")
            .document(chatId)
            .collection("messages")
            .orderBy("timestamp", Query.Direction.ASCENDING)
            .addSnapshotListener { snapshots, e ->
                if (e != null) return@addSnapshotListener
                messageList.clear()
                for (doc in snapshots!!) {
                    val message = doc.toObject(Message::class.java)
                    android.util.Log.d("DECRYPT", "text contains pipe: ${message.text.contains("|")}")
                    android.util.Log.d("DECRYPT", "parts by pipe: ${message.text.split("|", limit=3).size}")
                    android.util.Log.d("DECRYPT", "senderId: ${message.senderId}, currentUserId: $currentUserId")
                    android.util.Log.d("DECRYPT", "text parts count: ${message.text.split(":").size}")
                    android.util.Log.d("DECRYPT", "myPrivateKey empty: ${myPrivateKey.isEmpty()}")
                    // REPLACE:
                    val freshPrivateKey = if (myPrivateKey.isNotEmpty()) myPrivateKey else KeyStorageHelper.getPrivateKey(this) ?: ""
                    val displayMessage = if (message.type == "text"
                        && freshPrivateKey.isNotEmpty()
                        && message.encCiphertext.isNotEmpty()
                    ) {
                        val payload = if (message.senderId == currentUserId) {
                            // I sent this message — use sender slot first
                            EncryptedPayload(
                                encKeyForSender = message.encKeyForSender,
                                encKeyForRecipient = message.encKeyForRecipient,
                                iv = message.encIv,
                                ciphertext = message.encCiphertext
                            )
                        } else {
                            // I received this message — swap slots so recipient slot is tried first
                            EncryptedPayload(
                                encKeyForSender = message.encKeyForRecipient,
                                encKeyForRecipient = message.encKeyForSender,
                                iv = message.encIv,
                                ciphertext = message.encCiphertext
                            )
                        }
                        val decrypted = EncryptionHelper.decryptMessage(payload, freshPrivateKey)
                        message.copy(text = decrypted)
                    } else {
                        message
                    }
                    messageList.add(displayMessage)
                }
                adapter.notifyDataSetChanged()
                if (messageList.isNotEmpty()) {
                    recyclerMessages.scrollToPosition(messageList.size - 1)

                    // Update last message from receiver side
                    val lastMessage = messageList.last()
                    if (lastMessage.senderId != currentUserId) {
                        val chatRef = db.collection("chats").document(chatId)
                        val lastMessageText = if (lastMessage.type == "image")
                            "📷 Photo" else lastMessage.text
                        chatRef.get().addOnSuccessListener { doc ->
                            val savedTime = doc.getLong("lastMessageTime") ?: 0L
                            if (lastMessage.timestamp > savedTime) {
                                chatRef.update(
                                    "lastMessage", lastMessageText,
                                    "lastMessageTime", lastMessage.timestamp,
                                    "lastMessageSenderId", lastMessage.senderId
                                )
                            }
                        }
                    }
                }
                // Only mark as read if window is focused
                if (isWindowFocused) {
                    val lastMessage = messageList.lastOrNull()
                    if (lastMessage?.senderId != currentUserId &&
                        lastMessage?.messageId != lastSoundMessageId) {
                        lastSoundMessageId = lastMessage?.messageId ?: ""
                        playMessageSound()
                    }
                    markMessagesAsRead()
                }
            }
    }

    private fun listenToUserStatus(tvUserStatus: TextView) {
        db.collection("users").document(otherUserId)
            .addSnapshotListener { snapshot, e ->
                if (e != null || snapshot == null) return@addSnapshotListener
                val status = snapshot.getString("status") ?: "offline"
                if (status == "online") {
                    tvUserStatus.text = "online"
                    tvUserStatus.setTextColor(android.graphics.Color.parseColor("#CCFFFFFF"))
                } else {
                    val lastSeen = snapshot.getLong("lastSeen")
                    if (lastSeen != null) {
                        val timeStr = java.text.SimpleDateFormat(
                            "hh:mm a", java.util.Locale.getDefault()
                        ).format(java.util.Date(lastSeen))
                        tvUserStatus.text = "last seen at $timeStr"
                    } else {
                        tvUserStatus.text = "offline"
                    }
                    tvUserStatus.setTextColor(android.graphics.Color.parseColor("#AAFFFFFF"))
                }
            }
    }

    private fun sendNotificationToUser(recipientUid: String, messageText: String) {
        db.collection("users").document(recipientUid).get()
            .addOnSuccessListener { doc ->
                val activeChatId = doc.getString("activeChatId") ?: ""
                if (activeChatId == chatId) return@addOnSuccessListener

                val fcmToken = doc.getString("fcmToken") ?: return@addOnSuccessListener
                if (fcmToken.isEmpty()) return@addOnSuccessListener
                val senderName = currentUserName
                Thread {
                    try {
                        sendFcmV1Notification(fcmToken, senderName, messageText)
                    } catch (e: Exception) {
                        e.printStackTrace()
                    }
                }.start()
            }
    }

    private fun playMessageSound() {
        val mediaPlayer = android.media.MediaPlayer.create(this, R.raw.message_received)
        mediaPlayer.setAudioAttributes(

    private fun syncEncryptionKeys(onDone: () -> Unit) {
        val storedPrivateKey = KeyStorageHelper.getPrivateKey(this) ?: ""

        db.collection("users").document(currentUserId).get()
            .addOnSuccessListener { doc ->
                val firestorePublicKey = doc.getString("publicKey") ?: ""

                val keysAreValid = if (storedPrivateKey.isNotEmpty() && firestorePublicKey.isNotEmpty()) {
                    try {
                        val privateKeyBytes = android.util.Base64.decode(storedPrivateKey, android.util.Base64.NO_WRAP)
                        val privateKey = java.security.KeyFactory.getInstance("RSA")
                            .generatePrivate(java.security.spec.PKCS8EncodedKeySpec(privateKeyBytes))
                        val crtKey = privateKey as java.security.interfaces.RSAPrivateCrtKey
                        val spec = java.security.spec.RSAPublicKeySpec(crtKey.modulus, crtKey.publicExponent)
                        val derivedPublicKey = java.security.KeyFactory.getInstance("RSA").generatePublic(spec)
                        val derivedStr = android.util.Base64.encodeToString(
                            derivedPublicKey.encoded, android.util.Base64.NO_WRAP
                        )
                        val match = derivedStr == firestorePublicKey
                        android.util.Log.d("KEYS", "Keys MATCH: $match")
                        android.util.Log.d("KEYS", "Firestore pubkey last 20: ${firestorePublicKey.takeLast(20)}")
                        android.util.Log.d("KEYS", "Derived pubkey  last 20: ${derivedStr.takeLast(20)}")
                        match
                    } catch (e: Exception) {
                        android.util.Log.e("KEYS", "Key comparison error: ${e.message}")
                        false
                    }
                } else {
                    android.util.Log.e("KEYS", "storedPrivateKey empty: ${storedPrivateKey.isEmpty()}, firestorePublicKey empty: ${firestorePublicKey.isEmpty()}")
                    false
                }

                if (!keysAreValid) {
                    android.util.Log.w("KEYS", "Mismatch — regenerating key pair")
                    val (newPublicKey, newPrivateKey) = EncryptionHelper.generateKeyPair()
                    KeyStorageHelper.savePrivateKey(this, newPrivateKey)
                    myPrivateKey = newPrivateKey  // ← SET IT NOW, don't wait for loadEncryptionKeys
                    db.collection("users").document(currentUserId)
                        .update("publicKey", newPublicKey)
                        .addOnSuccessListener {
                            android.util.Log.d("KEYS", "New key pair saved to Firestore")
                            onDone()
                        }
                        .addOnFailureListener { e ->
                            android.util.Log.e("KEYS", "Failed to save new public key: ${e.message}")
                            onDone()
                        }
                } else {
                    android.util.Log.d("KEYS", "Keys valid — no regeneration needed")
                    onDone()  // ← proceed immediately
                }
            }
            .addOnFailureListener { e ->
                android.util.Log.e("KEYS", "Firestore fetch failed: ${e.message}")
                onDone()
            }
    }

    private fun loadEncryptionKeys(onReady: () -> Unit) {
        if (myPrivateKey.isEmpty()) {
            myPrivateKey = KeyStorageHelper.getPrivateKey(this) ?: ""
        }

        db.collection("users").document(currentUserId).get()
            .addOnSuccessListener { myDoc ->
                myPublicKey = myDoc.getString("publicKey") ?: ""

                db.collection("users").document(otherUserId).get()
                    .addOnSuccessListener { recipientDoc ->
                        recipientPublicKey = recipientDoc.getString("publicKey") ?: ""
                        android.util.Log.d("KEY_LOAD", "myPublicKey last 20: ${myPublicKey.takeLast(20)}")
                        android.util.Log.d("KEY_LOAD", "recipientPublicKey last 20: ${recipientPublicKey.takeLast(20)}")

                        android.util.Log.d("SLOT", "receiver myPublicKey last 20: ${myPublicKey.takeLast(20)}")
                        android.util.Log.d("SLOT", "receiver recipientPublicKey last 20: ${recipientPublicKey.takeLast(20)}")
                        onReady()
                    }
            }
    }
