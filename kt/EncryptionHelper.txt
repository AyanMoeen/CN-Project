package com.ayan.instantmessenger

import android.util.Base64
import javax.crypto.Cipher
import javax.crypto.KeyGenerator
import javax.crypto.spec.GCMParameterSpec
import javax.crypto.spec.SecretKeySpec
import java.security.KeyFactory
import java.security.KeyPairGenerator
import java.security.SecureRandom
import java.security.spec.PKCS8EncodedKeySpec
import java.security.spec.X509EncodedKeySpec

data class EncryptedPayload(
    val encKeyForSender: String = "",
    val encKeyForRecipient: String = "",
    val iv: String = "",
    val ciphertext: String = ""
)

object EncryptionHelper {

    fun generateKeyPair(): Pair<String, String> {
        val kpg = KeyPairGenerator.getInstance("RSA")
        kpg.initialize(2048)
        val kp = kpg.generateKeyPair()
        val pub = Base64.encodeToString(kp.public.encoded, Base64.NO_WRAP)
        val priv = Base64.encodeToString(kp.private.encoded, Base64.NO_WRAP)
        return Pair(pub, priv)
    }

    private fun rsaEncrypt(data: ByteArray, publicKeyString: String): ByteArray {
        val keyBytes = Base64.decode(publicKeyString, Base64.NO_WRAP)
        val publicKey = KeyFactory.getInstance("RSA")
            .generatePublic(X509EncodedKeySpec(keyBytes))
        val cipher = Cipher.getInstance("RSA/ECB/PKCS1Padding")
        cipher.init(Cipher.ENCRYPT_MODE, publicKey)
        return cipher.doFinal(data)
    }

    private fun rsaDecrypt(data: ByteArray, privateKeyString: String): ByteArray {
        val keyBytes = Base64.decode(privateKeyString, Base64.NO_WRAP)
        val privateKey = KeyFactory.getInstance("RSA")
            .generatePrivate(PKCS8EncodedKeySpec(keyBytes))
        val cipher = Cipher.getInstance("RSA/ECB/PKCS1Padding")
        cipher.init(Cipher.DECRYPT_MODE, privateKey)
        return cipher.doFinal(data)
    }

    fun encryptMessageForBoth(
        message: String,
        recipientPublicKeyString: String,
        senderPublicKeyString: String
    ): EncryptedPayload {
        // Generate AES-256 key
        val aesKey = KeyGenerator.getInstance("AES").also { it.init(256) }.generateKey()

        // Generate 12-byte IV for GCM
        val iv = ByteArray(12).also { SecureRandom().nextBytes(it) }

        // Encrypt message with AES-GCM
        val cipher = Cipher.getInstance("AES/GCM/NoPadding")
        cipher.init(Cipher.ENCRYPT_MODE, aesKey, GCMParameterSpec(128, iv))
        val ciphertext = cipher.doFinal(message.toByteArray(Charsets.UTF_8))

        // Encrypt AES key for both parties separately
        val encKeyForRecipient = rsaEncrypt(aesKey.encoded, recipientPublicKeyString)
        val encKeyForSender = rsaEncrypt(aesKey.encoded, senderPublicKeyString)

        android.util.Log.d("ENC_SIZE", "encKeyForRecipient bytes: ${encKeyForRecipient.size}")
        android.util.Log.d("ENC_SIZE", "encKeyForSender bytes: ${encKeyForSender.size}")

        return EncryptedPayload(
            encKeyForSender = Base64.encodeToString(encKeyForSender, Base64.NO_WRAP),
            encKeyForRecipient = Base64.encodeToString(encKeyForRecipient, Base64.NO_WRAP),
            iv = Base64.encodeToString(iv, Base64.NO_WRAP),
            ciphertext = Base64.encodeToString(ciphertext, Base64.NO_WRAP)
        )
    }

    fun decryptMessage(payload: EncryptedPayload, privateKeyString: String): String {
        return try {
            val encKeyForSender = Base64.decode(payload.encKeyForSender, Base64.NO_WRAP)
            val encKeyForRecipient = Base64.decode(payload.encKeyForRecipient, Base64.NO_WRAP)
            val iv = Base64.decode(payload.iv, Base64.NO_WRAP)
            val ciphertext = Base64.decode(payload.ciphertext, Base64.NO_WRAP)

            android.util.Log.d("DEC_SIZE", "encKeyForSender bytes: ${encKeyForSender.size}")
            android.util.Log.d("DEC_SIZE", "encKeyForRecipient bytes: ${encKeyForRecipient.size}")

            val aesKeyBytes = try {
                rsaDecrypt(encKeyForSender, privateKeyString)
                    .also { android.util.Log.d("DEC_SIZE", "Sender slot SUCCESS") }
            } catch (e: Exception) {
                android.util.Log.d("DEC_SIZE", "Sender slot FAILED: ${e.message}")
                try {
                    rsaDecrypt(encKeyForRecipient, privateKeyString)
                        .also { android.util.Log.d("DEC_SIZE", "Recipient slot SUCCESS") }
                } catch (e2: Exception) {
                    android.util.Log.d("DEC_SIZE", "Recipient slot FAILED: ${e2.message}")
                    return "[both slots failed]"
                }
            }

            val aesCipher = Cipher.getInstance("AES/GCM/NoPadding")
            aesCipher.init(
                Cipher.DECRYPT_MODE,
                SecretKeySpec(aesKeyBytes, "AES"),
                GCMParameterSpec(128, iv)
            )
            String(aesCipher.doFinal(ciphertext), Charsets.UTF_8)

        } catch (e: Exception) {
            "[decrypt error: ${e.message}]"
        }
    }

    fun selfTest(context: android.content.Context) {
        try {
            val (pub, priv) = generateKeyPair()
            val original = "Hello self test"
            val payload = encryptMessageForBoth(original, pub, pub)
            val decrypted = decryptMessage(payload, priv)
            android.util.Log.d("SELFTEST", "Original:  $original")
            android.util.Log.d("SELFTEST", "Decrypted: $decrypted")
            android.util.Log.d("SELFTEST", "SUCCESS: ${original == decrypted}")
        } catch (e: Exception) {
            android.util.Log.e("SELFTEST", "FAILED: ${e.message}")
        }
    }
}
